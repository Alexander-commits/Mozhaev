\section{Алгоритм Тонелли-Шенкса }

\begin{algorithm}
	
	\caption{Тонелли-Шенкса}
	\label{alg:AlgName}
	\textbf{Input:}\\
	$p$ - нечётное простое число; \\
	$n$ - целое число, являющееся квадратичным вычетом по модулю $p$. \\
	\textbf{Output:}\\
	Вычет $R$, удовлетворяющий сравнению $R^{2}\equiv n\: (mod\:p)$.
	
	\begin{algorithmic}[1]
		
		\State Выделить степени двойки из $p-1$, то есть положить $p-1= 2^{S}Q$, где Q-нечётно, {$S \geq 1$}. Если $S=1$, т.е. {$p\equiv 3\: (mod\:4)$}, решение имеет вид $R\equiv \pm\: n^{p+1 \over {4}}$. 
		\State Положить {$S \geq 2$}. Выбрать произвольный квадратичный невычет $z$, т.е. $({z \over {p}}) = -1$. Задать $c \equiv z^{Q}\: (mod\:p)$.
		\State Положить $R\equiv n^{Q+1 \over {2}}\:(mod\:p)$, $t\equiv n^{Q}\:(mod\:p)$, $M = S$. 
		\State \textbf{Пока} $t \neq 1$:
			\begin{algorithmic}[1]
				\State \textbf{Если} $t\equiv 1\:(mod\:p)$, то выход и ответ $R$.
				\State \textbf{Иначе для} $i = 1$ до $m$ искать $i$ такое, что $t^{2^{i}}\equiv 1\:(mod\:p)$ 
				\State Положить $b\equiv c^{2^{M-i-1}}\:(mod\:p)$, $R\equiv Rb\:(mod\:p)$, $t\equiv tb^{2}\:(mod\:p)$, $c\equiv b^{2}\:(mod\:p)$, $M = i$, вернуться к началу цикла.
			\end{algorithmic}	
	   \Statex Ответ: $R, p-R$.	
	\end{algorithmic}
\end{algorithm}
